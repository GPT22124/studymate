import fitz  # PyMuPDF
import openai
import faiss
import numpy as np

openai.api_key = 'YOUR_OPENAI_API_KEY'

# Function to extract text from PDF
def extract_text_from_pdf(pdf_path):
    doc = fitz.open(pdf_path)
    text = ""
    for page in doc:
        text += page.get_text()
    return text

# Function to split text into chunks (e.g. 500 chars)
def chunk_text(text, chunk_size=500):
    return [text[i:i+chunk_size] for i in range(0, len(text), chunk_size)]

# Get embedding vector for a text chunk
def get_embedding(text_chunk):
    response = openai.Embedding.create(
        input=text_chunk,
        model="text-embedding-ada-002"
    )
    return np.array(response['data'][0]['embedding'], dtype=np.float32)

# Build FAISS index from text chunks
def build_faiss_index(chunks):
    embeddings = [get_embedding(chunk) for chunk in chunks]
    dimension = len(embeddings[0])
    index = faiss.IndexFlatL2(dimension)
    index.add(np.vstack(embeddings))
    return index, embeddings

# Find most similar chunk index
def find_most_similar_chunk(query, index, chunks_embeddings, chunks):
    query_emb = get_embedding(query)
    distances, indices = index.search(np.array([query_emb]), k=1)
    best_idx = indices[0][0]
    return chunks[best_idx]

# Ask question to GPT using the most relevant chunk
def answer_question(question, context):
    prompt = f"Answer this question based on the context below:\n\nContext: {context}\n\nQuestion: {question}\nAnswer:"
    response = openai.Completion.create(
        engine="text-davinci-003",  # or gpt-4 if available
        prompt=prompt,
        max_tokens=150,
        temperature=0.2,
        n=1,
        stop=None,
    )
    return response['choices'][0]['text'].strip()

if __name__ == "__main__":
    pdf_path = "your_study_material.pdf"
    text = extract_text_from_pdf(pdf_path)
    chunks = chunk_text(text)

    print("Building index, please wait...")
    index, chunks_embeddings = build_faiss_index(chunks)
    print("Index built!")

    while True:
        question = input("\nAsk a question (or 'exit' to quit): ")
        if question.lower() == 'exit':
            break

        context = find_most_similar_chunk(question, index, chunks_embeddings, chunks)
        answer = answer_question(question, context)
        print("\nAnswer:", answer)

python
import fitz  # PyMuPDF for PDF reading
import openai
import faiss
import numpy as np

# Replace with your OpenAI API key
openai.api_key = "YOUR_OPENAI_API_KEY"

def extract_text_from_pdf(pdf_path):
    doc = fitz.open(pdf_path)
    text = ""
    for page in doc:
        text += page.get_text()
    return text

def chunk_text(text, max_chunk_size=500):
    # Splits text into chunks of max_chunk_size characters
    chunks = []
    start = 0
    while start < len(text):
        end = start + max_chunk_size
        chunks.append(text[start:end])
        start = end
    return chunks

def get_embedding(text):
    # Gets embedding vector for text using OpenAI
    response = openai.Embedding.create(
        input=text,
        model="text-embedding-ada-002"
    )
    return np.array(response['data'][0]['embedding'], dtype=np.float32)

def build_faiss_index(chunks):
    # Creates FAISS index of chunk embeddings for similarity search
    embeddings = [get_embedding(chunk) for chunk in chunks]
    dim = embeddings[0].shape[0]
    index = faiss.IndexFlatL2(dim)
    index.add(np.vstack(embeddings))
    return index, embeddings

def find_similar_chunk(question, index, chunks):
    # Finds the chunk most similar to the question
    question_emb = get_embedding(question)
    D, I = index.search(np.array([question_emb]), k=1)
    return chunks[I[0][0]]

def answer_question(question, context):
    # Uses GPT-3 to answer question based on context chunk
    prompt = f"Based on the following context, answer the question:\n\nContext:\n{context}\n\nQuestion:\n{question}\nAnswer:"
    response = openai.Completion.create(
        engine="text-davinci-003",
        prompt=prompt,
        max_tokens=150,
        temperature=0.2,
        n=1,
        stop=None
    )
    return response.choices[0].text.strip()

def main():
    pdf_path = input("Enter the path to your PDF file: ")
    print("Extracting text from PDF...")
    text = extract_text_from_pdf(pdf_path)
    
    print("Splitting text into chunks...")
    chunks = chunk_text(text)
    
    print("Building embedding index (this may take a moment)...")
    index, _ = build_faiss_index(chunks)
    
    print("Studymate is ready! Ask your questions (type 'exit' to quit).")
    while True:
        question = input("\nYour question: ")
        if question.lower() == "exit":
            print("Goodbye!")
            break
        context = find_similar_chunk(question, index, chunks)
        answer = answer_question(question, context)
        print(f"Answer: {answer}")

if __name__ == "__main__":
    main()

  • When you ask a question, it finds the most relevant chunk

  • Passes that chunk + your question to GPT to get a precise answer
